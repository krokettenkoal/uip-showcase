<script>
    let todos = [], newTodo = '';
    function addTodo(){
        todos.push(newTodo);
        todos = todos;
        newTodo = '';
    }
</script>

<ul>
    {#each todos as todo}
        <li>{todo}</li>
    {:else}
        <li>Nothing to do! ðŸ¥³</li>
    {/each}
</ul>
<input type="text" bind:value={newTodo} />
<button on:click={addTodo}>Add</button>
